#include <stdio.h>

long string_to_int(char *str) {
    long result = 0;
    int i = 0;

    while (str[i] != '\0' && str[i] != '\n' && str[i] != '\r') {
        if (str[i] >= '0' && str[i] <= '9') {
            result = result * 10 + (str[i] - '0');
        }
        i++;
    }
    return result;
}

int main() {
    FILE *file = fopen("in.txt", "r");
    char line[256];
    long total_sum = 0;

    if (file == NULL) {
        return 1;
    }

    while (fgets(line, sizeof(line), file)) {
        total_sum += string_to_int(line);
    }

    printf("%ld\n", total_sum);
    fclose(file);

    return 0;
}
