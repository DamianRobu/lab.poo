#include "NumberList.h"
#include <iostream>
using namespace std;
void NumberList::Init() {
    this->count = 0;
}

bool NumberList::Add(int x) {
    if (this->count >= 10) {
        return false;
    }
    this->numbers[this->count] = x;
    this->count++;
    return true;
}

void NumberList::Sort() {
    for (int i = 0; i < this->count - 1; i++) {
        for (int j = 0; j < this->count - i - 1; j++) {
            if (this->numbers[j] > this->numbers[j + 1]) {
                int temp = this->numbers[j];
                this->numbers[j] = this->numbers[j + 1];
                this->numbers[j + 1] = temp;
            }
        }
    }
}

void NumberList::Print() {
    cout << "Lista contine (" << this->count << " elemente): ";
    for (int i = 0; i < this->count; i++) {
        cout << this->numbers[i] << " ";
    }
    cout << endl;
}

#include "NumberList.h"

    NumberList myList;

    myList.Add(3);
    myList.Add(8);
    myList.Add(1);
    myList.Add(20);

    cout << "Dupa adaugare:" << endl;
    myList.Print();


    cout << "Dupa sortare:" <<endl;
    myList.Sort();
    myList.Print();

    for (int i = 0; i < 7; i++) {
        if (!myList.Add(100)) {
            cout << " Lista este plina, nu am adaugat elementul " << i + 6 << "!" << endl;
        }
    }

    return 0;
}
